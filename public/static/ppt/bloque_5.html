Procedimientos estadísticos
BLOQUE 5

KPIs Rápidos con Procedimientos Estadísticos

PROC MEANS para variables numéricas:

/* Estadísticas por categoría */
proc means data=work.ventas_cli_prod_vend 
           mean sum maxdec=2;
    class categoria;
    var monto_total;
    title 'KPIs de Monto por Categoría';
run; 
title;

/* Múltiples variables */
proc means data=work.ventas_cli_prod_vend 
           n mean stddev min max maxdec=2;
    var monto_total cantidad precio_unitario;
    title 'Estadísticas Descriptivas';
run; 
title;

PROC FREQ para categóricas:

/* Distribución de transacciones */
proc freq data=work.ventas_cli_prod_vend;
    tables categoria / nocum;
    title 'Ventas por Categoría';
run; 
title;

PROC REPORT ejecutivo:

/* Reporte KPI con totales */
proc report data=work.kpi_vendedor nowd;
    title "KPI por vendedor";
    columns vendedor n_ventas unidades 
            monto_total;
    define vendedor   / display "Vendedor";
    define n_ventas   / analysis sum 
                        "#Ventas";
    define unidades   / analysis sum 
                        "Unidades";
    define monto_total/ analysis sum 
                        format=dollar12.2 
                        "Monto total";
    rbreak after / summarize;
run; 
title;

PROC TABULATE (tablas cruzadas):

/* País vs Categoría */
proc tabulate data=work.ventas_cli_prod_vend 
              format=dollar12.2;
    title "Monto por País y Categoría";
    class pais categoria;
    var monto_total;
    table pais, categoria*monto_total*
                (sum mean);
run; 
title;
Los procedimientos estadísticos generan KPIs automáticamente sin programación manual. PROC MEANS, FREQ y REPORT son fundamentales.
Construcción de DataMarts
BLOQUE 5

Integración Progresiva de Fuentes y Construcción de DataMarts

Paso 1: Enriquecer con Clientes

/* Integración 1: Ventas + Clientes */
proc sql;
    create table work.ventas_cli as
    select v.*
         , c.id_char as id_cliente
         , c.nombre_completo, c.sexo_full
         , c.pais, c.email, c.email_dominio
         , c.vendedor as vendedor_asignado
    from work.ventas_validas v
    inner join work.clientes_nodup c
        on v.id_cliente_char = c.id_char;
quit;

Paso 2: Añadir Productos

/* Integración 2: + Productos */
proc sql;
    create table work.ventas_cli_prod as
    select vc.*
         , p.categoria
         , p.precio_unitario_num as precio_unitario
    from work.ventas_cli vc
    left join work.productos_nodup p
        on strip(vc.producto) = p.producto_char;
quit;

Paso 3: Completar con Vendedores

/* Integración 3: + Vendedores */
proc sql;
    create table work.ventas_cli_prod_vend as
    select vcp.*
         , v.vendedor_char as vendedor
         , v.region as region_vendedor
         , v.fecha_ingreso_d
         , v.objetivo_mensual
    from work.ventas_cli_prod vcp
    left join work.vendedores_nodup v
        on strip(vcp.vendedor_asignado) = v.vendedor_char;
quit;

DataMart RFM con Segmentación

/* DataMart final con análisis RFM */
proc sql;
    create table work.dm_clientes as
    select c.id_char as id, c.nombre_completo
         , k.n_transacciones as F  /* Frequency */
         , k.total_monto as M      /* Monetary */
         , intck('day', k.ultima_compra, today()) as R_dias
         , case when k.total_monto >= 5000 then 'VIP'
                when k.total_monto >= 2000 then 'Premium'
                else 'Regular' end as segmento length=12
    from work.clientes_nodup c
    left join work.kpi_cliente k on c.id_char = k.id_cliente;
quit;
La integración paso a paso permite verificar la calidad en cada etapa. El DataMart RFM es fundamental para segmentación de clientes.

Variables FIRST./LAST. por grupos:

/* Encontrar última compra por cliente */
proc sort data=work.ventas_prep out=work.ventas_ordenadas;
    by id_cliente fecha_venta_d;
run;

data work.ultima_compra;
    set work.ventas_ordenadas;
    by id_cliente fecha_venta_d;
    
    retain fecha_ultima_compra monto_ultima_compra;
    
    if first.id_cliente then do;
        fecha_ultima_compra = .;
        monto_ultima_compra = .;
    end;
    
    /* Actualizar con cada registro */
    fecha_ultima_compra = fecha_venta_d;
    monto_ultima_compra = monto_total;
    
    /* Solo salida en último registro del cliente */
    if last.id_cliente then output;
    
    format fecha_ultima_compra date9.;
    keep id_cliente fecha_ultima_compra monto_ultima_compra;
run;

Sistema de Auditoría y Monitoreo ETL

Macro de Auditoría Automatizada:

/* Sistema de auditoría completo */
%macro auditoria_etl(tabla_base=, tabla_resultado=, fecha_proceso=);
    proc sql;
        create table AUDIT.log_proceso_&fecha_proceso as
        select "&tabla_base" as tabla_origen length=50
             , "&tabla_resultado" as tabla_destino length=50
             , datetime() as timestamp_proceso format=datetime20.
             , (select count(*) from &tabla_base) as registros_origen
             , (select count(*) from &tabla_resultado) as registros_destino
             , calculated registros_destino - calculated registros_origen as diferencia
             , case when calculated diferencia = 0 then 'OK'
                    when calculated diferencia < 0 then 'PÉRDIDA'
                    else 'INCREMENTO' end as status length=12
        from (select 1 as dummy);
    quit;
    
    /* Reporte de problemas */
    %if &syserr ne 0 %then %do;
        %put ERROR: Fallo en auditoría de &tabla_base;
    %end;
%mend;

Validaciones específicas ETL:

/* Macro para detectar duplicados post-integración */
%macro check_duplicados(tabla=, key_fields=);
    proc sql;
        create table _temp_check as
        select &key_fields, count(*) as n_duplicados
        from &tabla
        group by &key_fields
        having count(*) > 1;
    quit;
    
    proc sql noprint;
        select count(*) into :n_problemas
        from _temp_check;
    quit;
    
    %if &n_problemas > 0 %then %do;
        %put WARNING: &n_problemas grupos con duplicados en &tabla;
        proc print data=_temp_check;
            title "Duplicados encontrados en &tabla";
        run;
    %end;
    %else %do;
        %put INFO: No se encontraron duplicados en &tabla;
    %end;
%mend;

/* Ejecutar auditorías */
