PROC SQL vs DATA Step: cuándo usar cada enfoque

PROC SQL (ideal para):

  • JOINs complejos con múltiples tablas
  • Agregaciones con GROUP BY y HAVING
  • Subconsultas y filtros avanzados
  • Sintaxis familiar (estándar SQL)
  • Operaciones en una sola instrucción

DATA Step (ideal para):

  • Control granular fila por fila
  • Lógica condicional compleja
  • Variables FIRST./LAST. por grupos
  • Transformaciones personalizadas
  • Optimización de memoria grandes volúmenes

Principio: PROC SQL para operaciones tipo base de datos, DATA step para control total y lógica compleja

PROC SQL para operaciones tipo base de datos, DATA step para lógica compleja y control total. Ambos enfoques son complementarios.

Tipos de JOINs en PROC SQL: casos de uso prácticos

INNER JOIN - Solo coincidencias:

/* Solo clientes con ventas */
proc sql;
    create table work.clientes_con_ventas as
    select c.id, c.nombre_completo, c.sexo_full
         , v.id_venta, v.producto, v.cantidad
         , v.monto_total, v.fecha_venta_d
    from work.clientes_prep as c
    inner join work.ventas_prep as v
        on c.id = v.id_cliente;
quit;

LEFT JOIN - Preservar tabla izquierda:

/* Todos los clientes, con/sin ventas */
proc sql;
    create table work.clientes_completo as
    select c.id, c.nombre_completo
         , v.id_venta, v.monto_total
    from work.clientes_prep as c
    left join work.ventas_prep as v
        on c.id = v.id_cliente;
quit;

FULL JOIN - Unión completa:

/* Detectar registros huérfanos */
proc sql;
    create table work.auditoria_completa as
    select coalesce(c.id, v.id_cliente) as id_master
         , c.id as id_en_clientes
         , v.id_cliente as id_en_ventas
         , c.nombre_completo, v.id_venta
    from work.clientes_prep c
    full join work.ventas_prep v
        on c.id = v.id_cliente;
quit;

/* Identificar casos problemáticos */
proc sql;
    create table work.ventas_huerfanas as
    select * from work.auditoria_completa
    where id_en_clientes is null;

Consideración: Indexar variables de unión para datasets grandes

INNER JOIN para análisis limpio, LEFT JOIN para cobertura completa, FULL JOIN para auditoría de integridad referencial.

Agregaciones y KPIs con GROUP BY y HAVING

KPIs básicos por cliente:

proc sql;
    create table work.kpi_cliente as
    select c.id, c.nombre_completo
         , count(v.id_venta) as n_transacciones
         , sum(coalesce(v.monto_total,0)) as total_monto
         , mean(v.monto_total) as ticket_promedio
         , min(v.fecha_venta_d) as primera_compra format=date9.
         , max(v.fecha_venta_d) as ultima_compra format=date9.
    from work.clientes_prep c
    left join work.ventas_prep v
        on c.id = v.id_cliente
    group by c.id, c.nombre_completo
    order by calculated total_monto desc;
quit;

Filtros con HAVING (sobre agregados):

/* Clientes VIP: >$3,000 en compras */
%let umbral_vip = 3000;
proc sql;
    create table work.clientes_vip as
    select * from work.kpi_cliente
    having total_monto >= &umbral_vip
       and n_transacciones >= 3;
quit;

KPIs por múltiples dimensiones:

/* KPIs por producto */
proc sql;
    create table work.kpi_producto as
    select producto
         , count(*) as n_ventas
         , sum(monto_total) as monto_total
         , mean(monto_total) as ticket_promedio
         , count(distinct id_cliente) as clientes_unicos
    from work.ventas_prep
    group by producto
    order by monto_total desc;
quit;
GROUP BY para consolidar por grupos, HAVING para filtrar agregados. Diferencia clave: WHERE filtra antes de agrupar, HAVING después.

Aplicación de formatos y uso de filtros

Formatos personalizados:

/* Crear formato personalizado */
PROC FORMAT;
    VALUE $sexoFmt
        'H' = 'Hombre'
        'M' = 'Mujer';
    VALUE rangoEdad
        0-17 = 'Menor'
        18-64 = 'Adulto'
        65-high = 'Senior';
RUN;

/* Aplicar formatos */
DATA work.datos;
    SET work.datos;
    FORMAT sexo $sexoFmt. edad rangoEdad.;
RUN;

Filtros (subsetting):

/* Filtro en DATA step */
DATA work.mayores;
    SET work.datos;
    IF edad >= 18;
RUN;

/* Filtro en PROC step */
PROC PRINT DATA=work.datos;
    WHERE sexo="M" AND salario > 50000;
RUN;

/* Filtro para casos sospechosos */
PROC PRINT DATA=work.datos;
    WHERE monto < 0;
    TITLE 'Registros con problemas';
RUN;
Diferencia entre formato permanente (DATA step) vs temporal (PROC step). WHERE vs IF.

BLOQUE 3

CASE WHEN y Variables Calculadas con CALCULATED

Segmentación con CASE WHEN:

/* Segmentación por nivel de gasto */
proc sql;
    create table work.segmento_cliente as
    select k.*
         , case
             when total_monto >= 5000 then 'VIP'
             when total_monto >= 2000 then 'Premium'
             when total_monto > 0 then 'Regular'
             else 'Sin Compras'
           end as segmento length=12
         , case
             when n_transacciones >= 10 then 'Muy Activo'
             when n_transacciones >= 5 then 'Activo'
             when n_transacciones >= 1 then 'Ocasional'
             else 'Inactivo'
           end as actividad length=12
    from work.kpi_cliente as k;
quit;

Uso de CALCULATED:

/* Filtrar por ticket promedio calculado */
proc sql;
    create table work.alto_ticket as
    select id, nombre_completo
         , n_transacciones
         , total_monto
         , case when n_transacciones > 0 
                then total_monto/n_transacciones 
                else . end as ticket_promedio
    from work.kpi_cliente
    having calculated ticket_promedio > 300;
quit;

Principio: CALCULATED permite referenciar columnas creadas en la misma consulta

Ventaja: Lógica condicional compleja directamente en SQL sin DATA steps adicionales

CASE WHEN es fundamental para crear categorizaciones dinámicas. CALCULATED evita subconsultas innecesarias.

DATA Step: SET/MERGE y Variables Automáticas FIRST./LAST.

MERGE - Combinación horizontal:

/* Agregar métricas por cliente */
proc sql;
    create table work.metricas_cliente as
    select id_cliente as id
         , count(*) as num_compras
         , sum(monto_total) as total_gastado
    from work.ventas_prep
    group by id_cliente;
quit;

/* Ordenar para MERGE */
proc sort data=work.clientes_prep; by id; run;
proc sort data=work.metricas_cliente; by id; run;

/* MERGE con indicadores IN= */
data work.cliente_completo;
    merge work.clientes_prep(in=tiene_datos_cliente) 
          work.metricas_cliente(in=tiene_ventas);
    by id;
    
    if tiene_datos_cliente and tiene_ventas then 
        tipo = 'Cliente Activo';
    else if tiene_datos_cliente and not tiene_ventas then 
        tipo = 'Cliente Sin Ventas';
    
    if tiene_datos_cliente; /* Solo clientes válidos */
run;

Variables FIRST./LAST. por grupos:

/* Encontrar última compra por cliente */
proc sort data=work.ventas_prep out=work.ventas_ordenadas;
    by id_cliente fecha_venta_d;
run;

data work.ultima_compra;
    set work.ventas_ordenadas;
    by id_cliente fecha_venta_d;
    
    retain fecha_ultima_compra monto_ultima_compra;
    
    if first.id_cliente then do;
        fecha_ultima_compra = .;
        monto_ultima_compra = .;
    end;
    
    /* Actualizar con cada registro */
    fecha_ultima_compra = fecha_venta_d;
    monto_ultima_compra = monto_total;
    
    /* Solo salida en último registro del cliente */
    if last.id_cliente then output;
    
    format fecha_ultima_compra date9.;
    keep id_cliente fecha_ultima_compra monto_ultima_compra;
run;

Gestión de duplicados con NODUPKEY

FIRST./LAST. son fundamentales para procesamiento por grupos. MERGE requiere ordenamiento previo pero ofrece máximo control.
